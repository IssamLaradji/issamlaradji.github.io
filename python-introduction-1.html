# print "annoying" if construction hour is between 7 (7:00 AM) 
# and 20 (8:00 PM)

def isAnnoying(hour):
    if (hour < 7) or (hour > 20):
        return "hour %d: Annoying" % hour
    else:
        return "hour %d: Not Annoying" % hour
    
print(isAnnoying(6))
print(isAnnoying(15))

hour 6: Annoying
hour 15: Not Annoying
